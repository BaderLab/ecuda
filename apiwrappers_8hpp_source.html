123 template<typename T>
124 inline cudaError_t cudaMemset2D( T* devPtr, const std::size_t pitch, const T& value, const std::size_t width, const std::size_t height ) {
125  std::vector< T, host_allocator<T> > v( width, value );
126  char* charPtr = reinterpret_cast<char*>(devPtr);
127  for( std::size_t i = 0; i < height; ++i, charPtr += pitch ) {
128  const cudaError_t rc = cudaMemcpy<T>( reinterpret_cast<T*>(charPtr), &v.front(), width, cudaMemcpyHostToDevice );
129  if( rc != cudaSuccess ) return rc;
130  }
131  return cudaSuccess;
132 }
