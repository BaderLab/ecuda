672  HOST void resize( const size_type newNumberRows, const size_type newNumberColumns, const size_type newNumberDepths, const value_type& value = value_type() ) {
673  if( number_rows() == newNumberRows and number_columns() == newNumberColumns and number_depths() == newNumberDepths ) return; // no resize needed
674  cube<value_type,allocator_type> newCube( newNumberRows, newNumberColumns, newNumberDepths, value );
675  for( size_type i = 0; i < std::min(newNumberRows,number_rows()); ++i ) {
676  for( size_type j = 0; j < std::min(newNumberColumns,number_columns()); ++j ) {

678  cudaMemcpy<value_type>(
