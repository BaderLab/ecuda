913 template<typename T,class Alloc>
914 HOST void vector<T,Alloc>::growMemory( size_type minimum ) {
915  if( m >= minimum ) return; // no growth neccessary
916  size_type m2 = m;
917  if( !m2 ) m2 = 1; // in case no memory is currently allocated
918  while( m2 < minimum ) m2 <<= 1;
919  // allocate larger chunk
920  device_ptr<value_type> newMemory( allocator.allocate( m2 ) );
921  // copy old data to new chunk
922  if( deviceMemory ) CUDA_CALL( cudaMemcpy<value_type>( newMemory.get(), deviceMemory.get(), m, cudaMemcpyDeviceToDevice ) );
923  deviceMemory = newMemory;
924  m = m2;
925 }
